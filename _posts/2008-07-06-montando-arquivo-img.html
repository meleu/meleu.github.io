prompt$ losetup -f arquivo.img
prompt$ fdisk -ul /dev/loop0 # ver o inicio da particao desejada
prompt$ mount arquivo.img /mnt/img -o ro,loop,offset=32256 # eg: 32256 = 63 * 512
